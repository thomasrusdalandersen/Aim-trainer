  <script>
    const gameArea = document.getElementById("gameArea");
    const hitsEl = document.getElementById("hits");
    const missesEl = document.getElementById("misses");
    const accuracyEl = document.getElementById("accuracy");
    const timeEl = document.getElementById("time");
    const scoreEl = document.getElementById("score");
    const messageEl = document.getElementById("message");
    const startBtn = document.getElementById("startBtn");
    const resetBtn = document.getElementById("resetBtn");
    const sizeSelect = document.getElementById("sizeSelect");
    const durationSelect = document.getElementById("durationSelect");

    let hits = 0;
    let misses = 0;
    let score = 0;
    let timeLeft = 30;
    let timer = null;
    let gameRunning = false;
    let currentTarget = null;

    function updateStats() {
      hitsEl.textContent = hits;
      missesEl.textContent = misses;
      scoreEl.textContent = score;
      const totalShots = hits + misses;
      const acc = totalShots === 0 ? 100 : Math.round((hits / totalShots) * 100);
      accuracyEl.textContent = acc;
      timeEl.textContent = timeLeft;
    }

    function clearTarget() {
      if (currentTarget) {
        currentTarget.remove();
        currentTarget = null;
      }
    }

    function createTarget() {
      clearTarget();

      const size = parseInt(sizeSelect.value, 10);
      const target = document.createElement("div");
      target.className = "target";
      target.style.width = size + "px";
      target.style.height = size + "px";

      const maxX = gameArea.clientWidth - size;
      const maxY = gameArea.clientHeight - size;

      const x = Math.random() * maxX;
      const y = Math.random() * maxY;

      target.style.left = x + "px";
      target.style.top = y + "px";

      const ring1 = document.createElement("div");
      ring1.className = "ring1";

      const ring2 = document.createElement("div");
      ring2.className = "ring2";

      const ring3 = document.createElement("div");
      ring3.className = "ring3";

      ring2.appendChild(ring3);
      ring1.appendChild(ring2);
      target.appendChild(ring1);

      target.addEventListener("click", function (e) {
        if (!gameRunning) return;
        e.stopPropagation();
        hits++;
        score += 100;
        updateStats();
        createTarget();
      });

      gameArea.appendChild(target);
      currentTarget = target;
    }

    function startGame() {
      clearInterval(timer);
      hits = 0;
      misses = 0;
      score = 0;
      timeLeft = parseInt(durationSelect.value, 10);
      gameRunning = true;
      updateStats();
      createTarget();
      messageEl.textContent = "Spillet er i gang.";

      timer = setInterval(() => {
        timeLeft--;
        updateStats();

        if (timeLeft <= 0) {
          endGame();
        }
      }, 1000);
    }

    function endGame() {
      clearInterval(timer);
      gameRunning = false;
      clearTarget();
      messageEl.textContent = `Ferdig! Score: ${score}, Treff: ${hits}, Bom: ${misses}, Accuracy: ${accuracyEl.textContent}%`;
    }

    function resetGame() {
      clearInterval(timer);
      gameRunning = false;
      hits = 0;
      misses = 0;
      score = 0;
      timeLeft = parseInt(durationSelect.value, 10);
      clearTarget();
      updateStats();
      messageEl.textContent = "Nullstilt. Trykk Start for å begynne.";
    }

    gameArea.addEventListener("click", function () {
      if (!gameRunning) return;
      misses++;
      updateStats();
    });

    startBtn.addEventListener("click", startGame);
    resetBtn.addEventListener("click", resetGame);

    resetGame();
  </script>
